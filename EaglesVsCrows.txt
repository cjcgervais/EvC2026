--[[
    BirdController — Client-side flight, input, camera, and FX handler
    This is where the spectacular flying dynamics come to life.
    Runs the FlightPhysics engine locally for zero-latency input response,
    then replicates the state to the server via HumanoidRootPart physics.
]]


-- Services
local RunService       = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local Players          = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local TweenService     = game:GetService("TweenService")


-- Modules
local GameConfig   = require(ReplicatedStorage:WaitForChild("GameConfig"))
local FlightPhysics = require(ReplicatedStorage:WaitForChild("FlightPhysics"))
local BirdBuilder  = require(ReplicatedStorage:WaitForChild("BirdBuilder"))


-- Remotes
local Remotes = ReplicatedStorage:WaitForChild("Remotes")


-- Player variables
local player = Players.LocalPlayer
local camera = workspace.CurrentCamera


-- State
local flightEngine = nil
local currentTeam = nil
local currentProfile = nil
local birdModel = nil
local isAlive = false
local thermalData = {}


-- Input State
local inputState = {
    pitch = 0,
    roll = 0,
    yaw = 0,
    flap = false,
    dive = false,
}


-- Camera State
local cameraState = {
    currentDistance = GameConfig.Camera.baseDistance,
    currentFOV = GameConfig.Camera.baseFOV,
}


--------------------------------------------------------------------------------
-- INPUT HANDLING
--------------------------------------------------------------------------------
local keyBindings = {
    [Enum.KeyCode.W] = { axis = "pitch", value = 1 },    -- Pitch down (nose down to dive/accelerate)
    [Enum.KeyCode.S] = { axis = "pitch", value = -1 },   -- Pitch up (nose up to climb)
    [Enum.KeyCode.A] = { axis = "roll", value = -1 },    -- Roll left
    [Enum.KeyCode.D] = { axis = "roll", value = 1 },     -- Roll right
    [Enum.KeyCode.Q] = { axis = "yaw", value = -1 },     -- Yaw left
    [Enum.KeyCode.E] = { axis = "yaw", value = 1 },      -- Yaw right
    [Enum.KeyCode.Space] = { axis = "flap", value = true },
    [Enum.KeyCode.LeftShift] = { axis = "dive", value = true },
}


UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    local binding = keyBindings[input.KeyCode]
    if binding then
        if binding.axis == "flap" or binding.axis == "dive" then
            inputState[binding.axis] = true
        else
            inputState[binding.axis] = binding.value
        end
    end


    -- Attack on Left Click
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        Remotes.AttackRequest:FireServer()
    end
end)


UserInputService.InputEnded:Connect(function(input, gameProcessed)
    local binding = keyBindings[input.KeyCode]
    if binding then
        if binding.axis == "flap" or binding.axis == "dive" then
            inputState[binding.axis] = false
        else
            -- Reset to 0 if the released key matches the current input direction
            if inputState[binding.axis] == binding.value then
                inputState[binding.axis] = 0
            end
        end
    end
end)


--------------------------------------------------------------------------------
-- THERMAL LOADING
--------------------------------------------------------------------------------
task.spawn(function()
    local getThermalsRF = ReplicatedStorage:FindFirstChild("Remotes"):FindFirstChild("GetThermals")
    if getThermalsRF then
        thermalData = getThermalsRF:InvokeServer() or {}
    end
end)


--------------------------------------------------------------------------------
-- CHARACTER SETUP
--------------------------------------------------------------------------------
local function OnCharacterAdded(character)
    -- Wait for the server to finish setting up the bird
    local rootPart = character:WaitForChild("HumanoidRootPart", 15)
    local humanoid = character:WaitForChild("Humanoid", 15)
    if not rootPart or not humanoid then return end


    -- Wait for team assignment from server
    local teamAssignment = Remotes.TeamAssigned.OnClientEvent:Wait()
    currentTeam = teamAssignment[1]
    currentProfile = teamAssignment[2]


    -- Find the bird model created by the server
    birdModel = character:FindFirstChildWhichIsA("Model")
    if not birdModel then
        birdModel = character:WaitForChild("EaglesBird", 5) or character:WaitForChild("CrowsBird", 5)
    end


    -- Initialize Flight Physics Engine
    flightEngine = FlightPhysics.new(currentProfile, rootPart.Position)
    flightEngine.orientation = rootPart.CFrame


    isAlive = true


    -- Disable default Roblox movement & camera
    humanoid.PlatformStand = true
    
    -- Set network ownership to client for smooth physics
    rootPart:SetNetworkOwner(player)


    -- Main Flight Loop
    RunService.RenderStepped:Connect(function(dt)
        if not isAlive or not flightEngine or not rootPart or not rootPart.Parent then
            isAlive = false
            return
        end


        ---------------------------------------------------
        -- 1. CALCULATE THERMAL FORCES
        ---------------------------------------------------
        local thermalForce = Vector3.new(0, 0, 0)
        for _, thermal in pairs(thermalData) do
            local flatPos = Vector3.new(rootPart.Position.X, 0, rootPart.Position.Z)
            local thermalPos = Vector3.new(thermal.position.X, 0, thermal.position.Z)
            local dist = (flatPos - thermalPos).Magnitude


            if dist < thermal.radius and rootPart.Position.Y > thermal.minHeight and rootPart.Position.Y < thermal.maxHeight then
                -- Closer to center = stronger thermal
                local centerFactor = 1 - (dist / thermal.radius)
                local heightFactor = 1 - math.abs((rootPart.Position.Y - (thermal.minHeight + thermal.maxHeight)/2) / (thermal.maxHeight - thermal.minHeight))
                thermalForce = thermalForce + Vector3.new(0, thermal.strength * centerFactor * heightFactor, 0)
            end
        end
        flightEngine:SetThermalForce(thermalForce)
        flightEngine:SetWind(GameConfig.Flight.WIND_BASE)


        ---------------------------------------------------
        -- 2. UPDATE FLIGHT PHYSICS
        ---------------------------------------------------
        flightEngine:Update(dt, inputState)


        ---------------------------------------------------
        -- 3. APPLY TO CHARACTER
        ---------------------------------------------------
        rootPart.CFrame = flightEngine.orientation
        rootPart.AssemblyLinearVelocity = flightEngine.velocity


        -- Remove any unwanted angular velocity from Roblox physics
        rootPart.AssemblyAngularVelocity = Vector3.new(0, 0, 0)


        ---------------------------------------------------
        -- 4. WING ANIMATION
        ---------------------------------------------------
        if birdModel then
            BirdBuilder.AnimateWings(
                birdModel,
                flightEngine.isFlapping,
                flightEngine.isDiving,
                flightEngine.flapPhase,
                dt
            )
        end


        ---------------------------------------------------
        -- 5. CAMERA UPDATE
        ---------------------------------------------------
        UpdateCamera(dt, rootPart)


        ---------------------------------------------------
        -- 6. COLLECTIBLE CHECK (Client-side trigger)
        ---------------------------------------------------
        local collectFolder = workspace:FindFirstChild("Collectibles")
        if collectFolder then
            for _, item in ipairs(collectFolder:GetChildren()) do
                if item:IsA("BasePart") and item.Transparency < 0.5 then
                    if (rootPart.Position - item.Position).Magnitude < 10 then
                        Remotes.CollectItem:FireServer(item.Name)
                    end
                end
            end
        end
        
        ---------------------------------------------------
        -- 7. HIT EFFECTS (Speed lines / FOV)
        ---------------------------------------------------
        -- FOV shift based on speed
        local targetFOV = GameConfig.Camera.baseFOV
        if flightEngine.isDiving then
            targetFOV = GameConfig.Camera.diveFOV
        elseif flightEngine.verticalSpeed > 30 then
            targetFOV = GameConfig.Camera.climbFOV
        end
        camera.FieldOfView = math.clamp(
            camera.FieldOfView + (targetFOV - camera.FieldOfView) * 0.1,
            50, 110
        )
    end)
end


player.CharacterAdded:Connect(OnCharacterAdded)
if player.Character then
    task.spawn(OnCharacterAdded, player.Character)
end

        -- Flash screen red

            end
        end

--------------------------------------------------------------------------------
-- SPECTACULAR CAMERA SYSTEM
--------------------------------------------------------------------------------
function UpdateCamera(dt, rootPart)
    camera.CameraType = Enum.CameraType.Scriptable


    local speed = flightEngine.speed
    local velocity = flightEngine.velocity


    -- Dynamic camera distance (pulls back at speed)
    local targetDistance = GameConfig.Camera.baseDistance + (speed * GameConfig.Camera.speedDistanceFactor)
    cameraState.currentDistance = cameraState.currentDistance + (targetDistance - cameraState.currentDistance) * 0.05


    -- Calculate camera offset based on bird orientation
    local backward = -rootPart.CFrame.LookVector
    local up = rootPart.CFrame.UpVector


    -- Offset: behind and slightly above
    local cameraOffset = (backward * cameraState.currentDistance) + (up * 6)


    -- Slight velocity-based offset (camera trails behind movement)
    if speed > 10 then
        local velNorm = velocity.Unit
        local velDot = velDot = backward:Dot(velNorm)
        cameraOffset = cameraOffset + (velNorm * velDot * speed * 0.05)
    end


    local targetCF = CFrame.new(rootPart.Position + cameraOffset, rootPart.Position)


    -- Bank tilt (camera rolls slightly with the bird)
    local rightVec = rootPart.CFrame.RightVector
    local bankAngle = math.asin(math.clamp(rightVec.Y, -1, 1))
    targetCF = targetCF * CFrame.Angles(0, 0, bankAngle * GameConfig.Camera.bankTiltFactor)


    -- Smooth follow
    camera.CFrame = camera.CFrame:Lerp(targetCF, GameConfig.Camera.followSmoothing * (1/dt * 0.001))


    -- Camera shake at high speed
    if speed > GameConfig.Camera.shakeAtSpeed then
        local shakeIntensity = ((speed - GameConfig.Camera.shakeAtSpeed) / 100) * GameConfig.Camera.shakeMagnitude
        camera.CFrame = camera.CFrame * CFrame.new(
            (math.random() - 0.5) * shakeIntensity,
            (math.random() - 0.5) * shakeIntensity,
            0
        )
    end
end
